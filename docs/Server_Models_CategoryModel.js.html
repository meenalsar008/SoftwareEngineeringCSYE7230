import mongoose from "mongoose";

/**
 * Category Model - Represents a product category in the online drugstore platform.
 * 
 * Fields:
 * @field name - The name of the category, a required string.
 * @field description - A brief description of the category, a required string.
 * 
 * Timestamps:
 * @field createdAt - Timestamp of category creation, automatically handled by Mongoose.
 * @field updatedAt - Timestamp of the last update to the category, automatically handled by Mongoose.
 */

const categorySchema = mongoose.Schema(
  {
    name: {
      type: String,
      required: true,
    },   
    description: {
      type: String,
      required: true,
    },
  },
  {
    timestamps: true,
  }
);

const Category = mongoose.model("Category", categorySchema);

export default Category;
