import axios from "axios";

/**
 * Fetches all categories from the server.
 * @route GET /api/categories
 * @group Categories - Operations about categories
 * @returns {Promise<Object[]>} A promise that resolves to an array of category objects.
 * @throws {Error} Throws an error if the server responds with an error status.
 */
export const fetchAllCategories = async () => {
  const response = await axios.get("/api/categories");
  return response?.data;
};

/**
 * Creates a new category.
 * @route POST /api/categories
 * @group Categories - Operations about categories
 * @param {Object} categoryForm - The category object to create.
 * @returns {Promise<Object>} A promise that resolves to the created category object.
 * @throws {Error} Throws an error if the server responds with an error status.
 */

export const createCategory = async (categoryForm) => {
    const response = await axios.post("/api/categories", categoryForm);
    return response?.data;
}

/**
 * Deletes a category by its ID.
 * @route DELETE /api/categories/{categoryId}
 * @group Categories - Operations about categories
 * @param {string} categoryId - The ID of the category to delete.
 * @returns {Promise} A promise that resolves when the category is deleted.
 * @throws {Error} Throws an error if the server responds with an error status.
 */

export const deleteCategory = async (categoryId) => {
    const response = await axios.delete(`/api/categories/${categoryId}`);
    return response;
}
