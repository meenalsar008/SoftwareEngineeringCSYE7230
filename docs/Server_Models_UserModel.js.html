/**
 * User Model - Represents a user of the online drugstore platform.
 * 
 * Fields:
 * @field name - The name of the user, a required string.
 * @field email - The email address of the user, used as a unique identifier for login. It's a required and unique string.
 * @field password - The password for the user account, stored in a hashed format for security, a required string.
 * @field isAdmin - Flag to indicate if the user has administrative privileges, default is false.
 * 
 * Methods:
 * @method matchPassword - A method to compare a given password with the stored hashed password.
 * 
 * Hooks:
 * @hook pre-save - A pre-save hook to hash the password before saving it to the database.
 * 
 * Timestamps:
 * @field createdAt - Timestamp of user account creation, automatically handled by Mongoose.
 * @field updatedAt - Timestamp of the last update to the user account, automatically handled by Mongoose.
 */
import mongoose from "mongoose";
import bcrypt from "bcryptjs";

const userSchema = mongoose.Schema(
  {
    name: {
      type: String,
      required: true,
    },
    email: {
      type: String,
      required: true,
      unique: true,
    },
    password: {
      type: String,
      required: true,
    },
    isAdmin: {
      type: Boolean,
      required: true,
      default: false,
    },
  },
  {
    timestamps: true,
  }
);

/**
 * Compares a given password with the user's stored password.
 * @param {String} enterPassword - Password to compare.
 * @returns {Promise<Boolean>} A promise that resolves with a boolean indicating if the password matches.
 */

// Login
userSchema.methods.matchPassword = async function (enterPassword) {
  return await bcrypt.compare(enterPassword, this.password);
};


/**
 * Hashes the user's password before saving to the database.
 * @hook
 * @param {Function} next - The next middleware function to call.
 */

// Register
userSchema.pre("save", async function (next) {
  if (!this.isModified("password")) {
    next();
  }
  const salt = await bcrypt.genSalt(10);
  this.password = await bcrypt.hash(this.password, salt);
});

const User = mongoose.model("User", userSchema);

export default User;
