import axios from "axios";


/**
 * @route GET /api/categories
 * @group Categories - Operations about categories
 * @returns {Object[]} 200 - An array of category objects.
 * @throws {Error} 500 - Internal server error.
 */
export const fetchAllCategories = async () => {
  const response = await axios.get("/api/categories");
  return response?.data;
};
/**
 * @route POST /api/categories
 * @group Categories - Operations about categories
 * @param {Object} req.body.required - The category object.
 * @param {string} req.body.name.required - The name of the category.
 * @param {string} req.body.description - The description of the category (optional).
 * @returns {Object} 201 - The created category object.
 * @throws {Error} 400 - Bad request if validation fails.
 * @throws {Error} 500 - Internal server error.
 */
export const createCategory = async (categoryForm) => {
    const response = await axios.post("/api/categories", categoryForm);
    return response?.data;
}

/**
 * @route DELETE /api/categories/{categoryId}
 * @group Categories - Operations about categories
 * @param {string} categoryId.required - The ID of the category to delete.
 * @returns {Object} 200 - Confirmation of deletion.
 * @throws {Error} 404 - If the category is not found.
 * @throws {Error} 500 - Internal server error.
 */

export const deleteCategory = async (categoryId) => {
    const response = await axios.delete(`/api/categories/${categoryId}`);
    return response;
}
